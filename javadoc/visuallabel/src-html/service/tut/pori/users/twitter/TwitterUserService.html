016package service.tut.pori.users.twitter;
017
018import service.tut.pori.users.Definitions;
019import core.tut.pori.http.Response;
020import core.tut.pori.http.Response.Status;
021import core.tut.pori.http.annotations.HTTPAuthenticationParameter;
022import core.tut.pori.http.annotations.HTTPMethodParameter;
023import core.tut.pori.http.annotations.HTTPService;
024import core.tut.pori.http.annotations.HTTPServiceMethod;
025import core.tut.pori.http.parameters.AuthenticationParameter;
026import core.tut.pori.http.parameters.StringParameter;
027import core.tut.pori.users.UserIdentity;
028
029/**
030 * Twitter User Service method definitions.
031 * 
032 * This class defines the APIs for registering a new account with the system and authentication using Twitter credentials, or more precisely, using the Twitter OAuth authentication. The methods also describe the means for granting the system authorization for accessing content of the Twitter account.
033 * 
034 * Creating an account or authenticating with Twitter credentials does not automatically grant the system persistent permissions for accessing the account contents.
035 * 
036 * Note that the Twitter User Service does not contain a method for removing an existing user account. An existing account can be removed using the User Service.
037 */
038@HTTPService(name=Definitions.SERVICE_USERS_TWITTER) 
039public class TwitterUserService {
040  /**
041   * This method is called by Twitter after user has either denied or granted the service permission to his/her Twitter account, and is not meant to be used directly by the end-users.
042   * 
043   * @param authenticatedUser 
044   * @param token OAuth token as provided by Twitter.
045   * @param verifier OAuth verifier generated by Twitter.
046   * @return response
047   */
048  @HTTPServiceMethod(name=Definitions.METHOD_OAUTH_LOGIN_CALLBACK)
049  public Response oAuthLoginCallback(
050      @HTTPAuthenticationParameter(required=false) AuthenticationParameter authenticatedUser,
051      @HTTPMethodParameter(name=Definitions.PARAMETER_OAUTH_TOKEN) StringParameter token,
052      @HTTPMethodParameter(name=Definitions.PARAMETER_OAUTH_VERIFIER) StringParameter verifier
053      )
054  {
055    return TwitterUserCore.processOAuthLoginCallback(authenticatedUser.getSession(), token.getValue(), verifier.getValue());
056  }
057  
058  /**
059   * This method is called by Twitter after user has either denied or granted the service permission to his/her Twitter account, and is not meant to be used directly by the end-users.
060   * 
061   * @param token OAuth token as provided by Twitter.
062   * @param verifier OAuth verifier generated by Twitter.
063   * @return response
064   */
065  @HTTPServiceMethod(name=Definitions.METHOD_OAUTH_AUTHORIZE_CALLBACK)
066  public Response oAuthAuthorizeCallback(
067      @HTTPMethodParameter(name=Definitions.PARAMETER_OAUTH_TOKEN) StringParameter token,
068      @HTTPMethodParameter(name=Definitions.PARAMETER_OAUTH_VERIFIER) StringParameter verifier
069      )
070  {
071    return TwitterUserCore.processOAuthAuthorizeCallback(token.getValue(), verifier.getValue());
072  }
073  
074  /**
075   * This method is called by Twitter after user has either denied or granted the service permission to his/her Twitter account, and is not meant to be used directly by the end-users.
076   * 
077   * @param token OAuth token as provided by Twitter.
078   * @param verifier OAuth verifier generated by Twitter.
079   * @return response
080   */
081  @HTTPServiceMethod(name=Definitions.METHOD_OAUTH_REGISTER_CALLBACK)
082  public Response oAuthRegisterCallback(
083      @HTTPMethodParameter(name=Definitions.PARAMETER_OAUTH_TOKEN) StringParameter token,
084      @HTTPMethodParameter(name=Definitions.PARAMETER_OAUTH_VERIFIER) StringParameter verifier
085      )
086  {
087    return TwitterUserCore.processOAuthRegisterCallback(token.getValue(), verifier.getValue());
088  }
089  
090  /**
091   * This method can be used to authenticate the user using Twitter credentials. The method redirects the user to Twitter's consent page, which will forward the user to the service's login OAuth callback method.
092   * 
093   * This method implements the Sign in with Twitter OAuth flow as defined by Twitter.
094   * 
095   * @param authenticatedUser
096   * @return response
097   */
098  @HTTPServiceMethod(name=Definitions.METHOD_LOGIN)
099  public Response login(@HTTPAuthenticationParameter(required=false) AuthenticationParameter authenticatedUser) {
100    if(UserIdentity.isValid(authenticatedUser.getUserIdentity())){
101      return new Response(Status.BAD_REQUEST, "Already logged in. Please logout first.");
102    }
103    return TwitterUserCore.createLoginRedirection();
104  }
105  
106  /**
107   * This method can be used to register a new user using a Twitter account. The method redirects the user to Twitter's consent page, which will forward the user to the service's register OAuth callback method. The twitter profile details will be used for the new user registration.
108   * 
109   * This method implements the Sign in with Twitter OAuth flow as defined by Twitter.
110   * 
111   * The Twitter account is assumed to be personal, and an attempt to use the same account to create multiple accounts will result in an error.
112   * 
113   * @return response
114   */
115  @HTTPServiceMethod(name=Definitions.METHOD_REGISTER)
116  public Response register() {
117    return TwitterUserCore.createRegisterRedirection();
118  }
119  
120  /**
121   * This method can be used to grant the service access to the authenticated user's Twitter account. The method redirects the user to Twitter's consent page, which will forward the user to the service's authorization OAuth callback method.
122   * 
123   * This method implements the Sign in with Twitter OAuth flow as defined by Twitter.
124   * 
125   * The Twitter account is assumed to be personal, and an attempt to authorize the same Twitter account to multiple users will result in an error.
126   * 
127   * @param authenticatedUser
128   * @param redirectUri The final target, where the user should be redirected to after successful authorization. If not given, a default status message will be shown.
129   * @return response
130   */
131  @HTTPServiceMethod(name=Definitions.METHOD_OAUTH_AUTHORIZATION_REDIRECT)
132  public Response authorize(
133      @HTTPAuthenticationParameter AuthenticationParameter authenticatedUser,
134      @HTTPMethodParameter(name=Definitions.PARAMETER_TWITTER_REDIRECT_URI, required=false) StringParameter redirectUri
135      )
136  {
137    return TwitterUserCore.createAuthorizationRedirection(authenticatedUser.getUserIdentity(), redirectUri.getValue());
138  }
139  
140  /**
141   * This method allows a user to revoke the previously given authorization permissions. Note that calling this method does not prevent user from authenticating (logging in) using the credentials, nor will it remove the user account from the system, but calling this method will prevent the system from accessing the user's content stored in the Twitter account. 
142   * 
143   * In practice this means, that no synchronization can be performed for the account until authorization is restored. <i>All</i> previously synchronized content will be removed from the system.
144   * 
145   * @param authenticatedUser
146   * @return response
147   */
148  @HTTPServiceMethod(name=Definitions.METHOD_UNAUTHORIZE)
149  public Response unauthorize(@HTTPAuthenticationParameter AuthenticationParameter authenticatedUser) {
150    return TwitterUserCore.removeAuthorization(authenticatedUser.getUserIdentity());
151  }
152}
