040/**
041 * DAO for handling authorization tokens for Google User Service.
042 */
043public class GoogleUserDAO extends SQLDAO{
044  private static final Logger LOGGER = Logger.getLogger(GoogleUserDAO.class);
045  /* tables */
046  private static final String TABLE_USERS_GOOGLE = DATABASE+".users_google";
047  /* columns */
048  private static final String COLUMN_ACCESS_TOKEN = "access_token";
049  private static final String COLUMN_ACCESS_TOKEN_EXPIRES = "access_token_expires";
050  private static final String COLUMN_NONCE = "nonce";
051  private static final String COLUMN_NONCE_EXPIRES = "nonce_expires";
052  private static final String COLUMN_REFRESH_TOKEN = "refresh_token";
053  private static final String COLUMN_TOKEN_TYPE = "token_type";
054  /* sql scripts */
055  private static final String SQL_COUNT_USER_ID = "SELECT "+COLUMN_COUNT+" FROM "+TABLE_USERS_GOOGLE+" WHERE "+COLUMN_USER_ID+"=?";
056  private static final int[] SQL_COUNT_USER_ID_SQL_TYPES = new int[]{SQLType.LONG.toInt()};
057  
058  private static final String SQL_INSERT_NONCE = "INSERT INTO "+TABLE_USERS_GOOGLE+" ("+COLUMN_USER_ID+","+COLUMN_NONCE+","+COLUMN_NONCE_EXPIRES+","+COLUMN_ROW_CREATED+") VALUES (?,?,?,NOW())";
059  private static final int[] SQL_INSERT_NONCE_SQL_TYPES = new int[]{SQLType.LONG.toInt(),SQLType.STRING.toInt(),SQLType.LONG.toInt()};
060  
061  private static final String SQL_REMOVE_NONCE = "UPDATE "+TABLE_USERS_GOOGLE+" SET "+COLUMN_NONCE+"=NULL WHERE "+COLUMN_NONCE+"=? LIMIT 1";
062  private static final int[] SQL_REMOVE_NONCE_SQL_TYPES = new int[]{SQLType.STRING.toInt()};
063  
064  private static final String SQL_REMOVE_TOKEN = "DELETE FROM "+TABLE_USERS_GOOGLE+" WHERE "+COLUMN_USER_ID+"=?";
065  private static final int[] SQL_REMOVE_TOKEN_SQL_TYPES = new int[]{SQLType.LONG.toInt()};
066  
067  private static final String SQL_SELECT_USER_ID = "SELECT "+COLUMN_COUNT+", "+COLUMN_USER_ID+" FROM "+TABLE_USERS_GOOGLE+" WHERE "+COLUMN_NONCE+"=? AND "+COLUMN_NONCE_EXPIRES+">? LIMIT 1";
068  private static final int[] SQL_SELECT_USER_ID_SQL_TYPES = new int[]{SQLType.STRING.toInt(),SQLType.LONG.toInt()};
069  
070  private static final String SQL_SELECT_TOKEN = "SELECT "+COLUMN_COUNT+", "+COLUMN_ACCESS_TOKEN+", "+COLUMN_ACCESS_TOKEN_EXPIRES+","+COLUMN_REFRESH_TOKEN+","+COLUMN_TOKEN_TYPE+" FROM "+TABLE_USERS_GOOGLE+" WHERE "+COLUMN_USER_ID+"=? AND ("+COLUMN_ACCESS_TOKEN_EXPIRES+">? OR "+COLUMN_REFRESH_TOKEN+" IS NOT NULL) LIMIT 1";
071  private static final int[] SQL_SELECT_TOKEN_SQL_TYPES = new int[]{SQLType.LONG.toInt(),SQLType.LONG.toInt()}; 
072  
073  /** tokens generated by google should be unique, so the unique column constraint on access token should never match on ON DUPLICATE KEY UPDATE for two different users, the match will be solely done on user id */
074  private static final String SQL_SET_TOKEN = "INSERT INTO "+TABLE_USERS_GOOGLE+" ("+COLUMN_USER_ID+","+COLUMN_ACCESS_TOKEN+","+COLUMN_ACCESS_TOKEN_EXPIRES+","+COLUMN_REFRESH_TOKEN+","+COLUMN_TOKEN_TYPE+","+COLUMN_ROW_CREATED+") VALUES (?,?,?,?,?,NOW()) ON DUPLICATE KEY UPDATE "+COLUMN_ACCESS_TOKEN+"=VALUES("+COLUMN_ACCESS_TOKEN+"), "+COLUMN_ACCESS_TOKEN_EXPIRES+"=VALUES("+COLUMN_ACCESS_TOKEN_EXPIRES+"), "+COLUMN_REFRESH_TOKEN+"=VALUES("+COLUMN_REFRESH_TOKEN+"), "+COLUMN_TOKEN_TYPE+"=VALUES("+COLUMN_TOKEN_TYPE+")";
075  private static final int[] SQL_SET_TOKEN_SQL_TYPES = new int[]{SQLType.LONG.toInt(),SQLType.STRING.toInt(),SQLType.LONG.toInt(),SQLType.STRING.toInt(),SQLType.STRING.toInt()};
076  
077  private static final String SQL_UPDATE_NONCE = "UPDATE "+TABLE_USERS_GOOGLE+" SET "+COLUMN_NONCE+"=?,"+COLUMN_NONCE_EXPIRES+"=? WHERE "+COLUMN_USER_ID+"=?";
078  private static final int[] SQL_UPDATE_NONCE_SQL_TYPES = new int[]{SQLType.STRING.toInt(),SQLType.LONG.toInt(),SQLType.LONG.toInt()};
079  
080  @Autowired
081  private UserDAO _userDAO = null;
082  
083  /**
084   * 
085   * @param googleUserId
086   * @param token
087   * @param userId
088   * @return true on success
089   */
090  public boolean setToken(final String googleUserId, final OAuth2Token token, final UserIdentity userId){
091    if(!UserIdentity.isValid(userId) || !token.isValid() || StringUtils.isBlank(googleUserId)){
092      LOGGER.debug("Invalid userId or token.");
093      return false;
094    }
095
096    return getTransactionTemplate().execute(new TransactionCallback<Boolean>() {
097
098      @Override
099      public Boolean doInTransaction(TransactionStatus status) {
100        JdbcTemplate t = getJdbcTemplate();
101        
102        String oldGoogleUserId = getGoogleUserId(userId);
103        if(oldGoogleUserId == null){
104          LOGGER.debug("Creating a new external account connection for user, id: "+userId.getUserId()+", google user id: "+googleUserId);
105          _userDAO.insertExternalAccountConnection(new ExternalAccountConnection(googleUserId, UserServiceType.GOOGLE), userId);
106        }else if(!oldGoogleUserId.equals(googleUserId)){  // do not allow changing the google user, this may cause undefined behavior on other parts of the system
107          LOGGER.warn("The given google user Id "+googleUserId+" did not match the old google user id: "+oldGoogleUserId);
108          return false;
109        }
110        
111        t.update(SQL_SET_TOKEN, new Object[]{userId.getUserId(),token.getAccessToken(),token.getExpires().getTime(),token.getRefreshToken(),token.getType()}, SQL_SET_TOKEN_SQL_TYPES);
112        return true;
113      }
114    });
115  }
116  
117  /**
118   * 
119   * @param userId non-null, valid userid
120   * @return google user id for the given userId, if known by the system, otherwise returns null
121   */
122  public String getGoogleUserId(UserIdentity userId){
123    ExternalAccountConnection connection = _userDAO.getExternalAccountConnection(userId, UserServiceType.GOOGLE);
124    return (connection == null ? null : connection.getExternalId());
125  }
126  
127  /**
128   * Removes the given nonce if such exists
129   * 
130   * @param nonce
131   */
132  public void removeNonce(String nonce){
133    if(getJdbcTemplate().update(SQL_REMOVE_NONCE, new Object[]{nonce}, SQL_REMOVE_NONCE_SQL_TYPES) < 1){
134      LOGGER.debug("No nonce was cleared.");
135    }else{
136      LOGGER.debug("Nonce cleared.");
137    }
138  }
139  
140  /**
141   * 
142   * @param userId
143   * @return token for the given user or null if none available
144   */
145  public OAuth2Token getToken(UserIdentity userId){
146    if(!UserIdentity.isValid(userId)){
147      LOGGER.debug("Invalid userId.");
148      return null;
149    }
150  
151    OAuth2Token token = new OAuth2Token();
152    Map<String, Object> row = getJdbcTemplate().queryForMap(SQL_SELECT_TOKEN, new Object[]{userId.getUserId(), System.currentTimeMillis()}, SQL_SELECT_TOKEN_SQL_TYPES);
153    for(Entry<String,Object> entry : row.entrySet()){
154      String column = entry.getKey();
155      switch(column){
156        case COLUMN_ACCESS_TOKEN:
157          token.setAccessToken((String) entry.getValue());
158          break;
159        case COLUMN_ACCESS_TOKEN_EXPIRES:
160          token.setExpires(new Date((Long) entry.getValue()));
161          break;
162        case COLUMN_REFRESH_TOKEN:
163          token.setRefreshToken((String) entry.getValue());
164          break;
165        case COLUMN_TOKEN_TYPE:
166          token.setType((String) entry.getValue());
167          break;
168        default:
169          if(checkCountColumn(column, entry.getValue()) < 1){ // if we have only one column, check what it is (should be count)
170            LOGGER.debug("Unknown column name, or no results.");
171            return null;
172          }
173          break;
174      }
175    }
176    return token;
177  }
178  
179  /**
180   * This generates a new nonce from the given UserId, replacing the old one if one was present
181   * 
182   * @param userId
183   * @return new nonce generated for the given userId
184   */
185  public String generateNonce(UserIdentity userId){
186    if(!UserIdentity.isValid(userId)){
187      LOGGER.debug("Invalid userId.");
188      return null;
189    }
190    Nonce nonce = NonceUtils.generateNonce(); // in theory, it is extremely unlikely that the generated nonce will collide with previously generated one, but...
191    if(recordExists(userId.getUserId())){ // ...just in case do not use INSERT ... ON DUPLICATE KEY UPDATE for updating the nonce (as nonce itself is an unique column)
192      getJdbcTemplate().update(SQL_UPDATE_NONCE, new Object[]{nonce.getNonce(),nonce.getExpires(),userId.getUserId()}, SQL_UPDATE_NONCE_SQL_TYPES); // this throw an exception on duplicate nonce if one ever occurs
193    }else{
194      getJdbcTemplate().update(SQL_INSERT_NONCE, new Object[]{userId.getUserId(),nonce.getNonce(),nonce.getExpires()}, SQL_INSERT_NONCE_SQL_TYPES); // this throw an exception on duplicate nonce if one ever occurs
195    }
196    return nonce.getNonce();
197  }
198  
199  /**
200   * remove token from the given userIdentity if one exists
201   * 
202   * @param userId
203   */
204  public void removeToken(UserIdentity userId){
205    if(!UserIdentity.isValid(userId)){
206      LOGGER.debug("Invalid userId.");
207      return;
208    }
209    Long id = userId.getUserId();
210    if(getJdbcTemplate().update(SQL_REMOVE_TOKEN,new Object[]{id},SQL_REMOVE_TOKEN_SQL_TYPES) < 1){
211      LOGGER.debug("No tokens removed for userId: "+id);
212    }else{
213      LOGGER.debug("Token removed for userId: "+id);
214    }
215  }
216  
217  /**
218   * 
219   * @param nonce
220   * @return owner of the given nonce, or null if no such nonce exists or the nonce has expired
221   */
222  public UserIdentity getUser(String nonce){
223    Long userId = (Long) getJdbcTemplate().queryForMap(SQL_SELECT_USER_ID, new Object[]{nonce,System.currentTimeMillis()}, SQL_SELECT_USER_ID_SQL_TYPES).get(COLUMN_USER_ID);
224    return (userId == null ? null : new UserIdentity(userId));
225  }
226  
227  /**
228   * 
229   * @param userId
230   * @return true if a record exists for the given user id
231   */
232  private boolean recordExists(Long userId){
233    return (getJdbcTemplate().queryForObject(SQL_COUNT_USER_ID, new Object[]{userId}, SQL_COUNT_USER_ID_SQL_TYPES, Long.class) > 0);
234  }
