182  /**
183   * The grant type of the OAuth2 token.
184   */
185  public enum OAuth2GrantType{
186    /** oauth2 authorization code (<a href="https://tools.ietf.org/html/rfc6749#section-1.3.1">OAuth 2 authorization code</a>) */
187    authorization_code,
188    /** oauth2 refresh token (<a href="https://tools.ietf.org/html/rfc6749#section-1.5">OAuth 2 refresh token</a>) */
189    refresh_token;
190    
191    /**
192     * 
193     * @return this grant type as string
194     */
195    public String toOAuth2GrantType(){
196      return name();
197    }
198  } // enum OAuth2GrantType
