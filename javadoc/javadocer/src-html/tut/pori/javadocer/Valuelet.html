016package tut.pori.javadocer;
017
018import java.util.Map;
019
020import com.sun.javadoc.Tag;
021import com.sun.tools.doclets.formats.html.markup.RawHtml;
022import com.sun.tools.doclets.internal.toolkit.Content;
023import com.sun.tools.doclets.internal.toolkit.taglets.TagletWriter;
024import com.sun.tools.doclets.internal.toolkit.taglets.ValueTaglet;
025
026/**
027 * Extends the default \@value tag by removing the unnecessary "" around the value, has no effect on the RESTlet functionality.
028 * 
029 * Note that this uses Java's internal classes, see the commented code for implementations for different Java versions.
030 * 
031 * Optionally, simply using other name for the tag than \@value for values, would remove the need to use internal classes.
032 * 
033 * For usage instructions, follow the Oracle's guide at <a href="http://docs.oracle.com/javase/7/docs/technotes/guides/javadoc/taglet/overview.html">Taglet Overview</a>.
034 */
035public class Valuelet extends ValueTaglet {
036        
037        /*
038         *  for JAVA 7 
039         *  
040         *  Requires: import com.sun.tools.doclets.internal.toolkit.taglets.TagletOutput;
041         *
042        @Override
043        public TagletOutput getTagletOutput(Tag tag, TagletWriter writer) {
044                TagletOutput output = super.getTagletOutput(tag, writer);
045                String value = output.toString();
046                value = value.replace(">\"", ">"); // remove " " around string values in the href ...a>"value"</a>
047                value = value.replace("\"<", "<");
048                output.setOutput(value);
049                return output;
050        }*/
051
052        /**
053         * For JAVA 8
054         * 
055         * Requires: import com.sun.tools.doclets.internal.toolkit.Content
056         */
057        @Override
058        public Content getTagletOutput(Tag tag, TagletWriter writer) {
059                String value = super.getTagletOutput(tag, writer).toString();
060                value = value.replace(">\"", ">"); // remove " " around string values in the href ...a>"value"</a>
061                value = value.replace("\"<", "<");
062                return new RawHtml(value); // create new content with the replaced values
063        }
064
065        /**
066     * Register this Taglet.
067     * @param tagletMap the map to register this tag to.
068     */
069    @SuppressWarnings({ "rawtypes", "unchecked" })
070        public static void register(Map tagletMap) {
071        Valuelet tag = new Valuelet();
072        tagletMap.put(tag.getName(), tag);
073    }
074}
